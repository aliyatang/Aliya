import getpass

def q_w_r(question, answer):
    print(question)
    res = input(question)
    if res == answer:
        print(res + " is correct! Nice job.")
        points = 1
    else:
        print("Sorry that's not right. Next question!")
        points = 0
    return points

questions = 4
