What is a conditional?

x = input("Enter a decimal number:")

## Decimal to Binary
numsBin = []
def convertBin(x):
    if x.isnumeric():
        x = int(x)
        while x/2 != 0:
            rem = x % 2
            numsBin.append(rem)
            x = int(x/2)
        numsBin.reverse()
        binary = " "
        for i in numsBin:
            binary = binary + str(i)
        return binary
    else:
        print("Not a decimal number.")
        quit

print(str(x) + " to binary: " + convertBin(x))


## Decimal to Octal
numsOct = []
def convertOct(x):
    if x.isnumeric():
        x = int(x)
        if x < 8:
            octal = x
            return octal
        else:
            while x/8 != 0:
                rem = x % 8
                numsOct.append(rem)
                x = int(x/8)
            numsOct.reverse()
            octal = " "
            for i in numsOct:
                octal = octal + str(i)
            return octal
    else:
        print("Not a decimal number.")
        quit

print(str(x) + " to octal: " + str(convertOct(x)))
